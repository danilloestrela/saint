#!/bin/bash
# Main commands for ${newProjName}
${newProjNameForBash}_main_commands() {
  lastCompleteCommand="$1 $2"

  case $2 in
    code) ${newProjNameForBash}_code ${lastCompleteCommand} ${@:3};;
    goto) ${newProjNameForBash}_goto ${lastCompleteCommand} ${@:3};;
    help) ${newProjNameForBash}_help;;
    install) ${newProjNameForBash}_install  ${lastCompleteCommand} ${@:3};;
    run) ${newProjNameForBash}_run  ${lastCompleteCommand} ${@:3};;
    update-help) ${newProjNameForBash}_update_help;;
    "") saint_empty ${lastCompleteCommand};;
    *) saint_invalid_command $1 $2;;
  esac

}
